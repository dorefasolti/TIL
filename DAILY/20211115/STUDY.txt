

/*
    LIKE
*/
SELECT * FROM EMPLOYEE
WHERE EMP_NAME LIKE '전%';


SELECT * FROM EMPLOYEE
WHERE EMP_NAME LIKE '%하%';


SELECT * FROM EMPLOYEE
WHERE SUBSTR(PHONE, 4, 1) LIKE 9;

SELECT * FROM EMPLOYEE
WHERE PHONE LIKE '___9%';


SELECT * FROM EMPLOYEE
WHERE EMP_NAME LIKE '%지%';

SELECT * FROM EMPLOYEE
WHERE EMP_NAME NOT LIKE '%지%';

SELECT * FROM EMPLOYEE
WHERE EMP_NAME LIKE '__연';

SELECT * FROM EMPLOYEE
WHERE EMP_NAME LIKE '%연';

SELECT * FROM EMPLOYEE
WHERE SUBSTR(PHONE, 1, 3) NOT LIKE '010';

SELECT * FROM EMPLOYEE
WHERE PHONE NOT LIKE '010%';

SELECT * FROM DEPARTMENT
WHERE DEPT_TITLE LIKE '해외영업%';


/*

    IS NULL

*/
SELECT * FROM EMPLOYEE
WHERE BONUS IS NULL;

SELECT * FROM EMPLOYEE
WHERE BONUS IS NOT NULL;

SELECT * FROM EMPLOYEE
WHERE MANAGER_ID IS NULL;

SELECT * FROM EMPLOYEE
WHERE MANAGER_ID IS NULL AND DEPT_CODE IS NULL;

SELECT * FROM EMPLOYEE
WHERE BONUS IS NOT NULL AND DEPT_CODE IS NULL;


/*

    IN

*/
SELECT * FROM EMPLOYEE
WHERE DEPT_CODE IN ('D6', 'D5');
SELECT * FROM EMPLOYEE
WHERE DEPT_CODE NOT IN('D6', 'D5');


/*
    문자열과 관련된 함수

*/
        
SELECT LENGTH('오라클'), LENGTHB('오라클')
FROM DUAL;


SELECT EMAIL, LENGTH(EMAIL), LENGTHB(EMAIL)
FROM EMPLOYEE;


/*
    INSTR

*/
SELECT EMP_NAME, INSTR(EMP_NAME, '김')
FROM EMPLOYEE;

/*

    SUBSTR
    
*/

SELECT DECODE(SUBSTR(EMP_NO, 8, 1), 1, '남', 2, '여')
FROM EMPLOYEE;

SELECT DECODE(SUBSTR(EMP_NO, 8, 1), 1, '남')
FROM EMPLOYEE
WHERE SUBSTR(EMP_NO, 8, 1) IN (1, 3);

SELECT DECODE(SUBSTR(EMP_NO, 8, 1), 2, '여', 4, '여')
FROM EMPLOYEE
WHERE SUBSTR(EMP_NO, 8, 1) IN (2, 4);

SELECT SUBSTR(EMAIL, 1, INSTR(EMAIL, '@')-1 ) 
FROM EMPLOYEE;

/*

    LPAD/RPAD

*/

SELECT EMP_NAME, RPAD(SUBSTR(EMP_NO, 1, 8), 14, '*')
FROM EMPLOYEE;

/*
    
    TRIM/LTRIM/RTRIM
    
*/

SELECT LTRIM('          K         I                 M')
FROM DUAL;

SELECT LTRIM('0101234567860000', '0')
FROM DUAL;

SELECT RTRIM('ABCKASDLFKJAAAAAAA', 'A')
FROM DUAL;

SELECT TRIM('Z' FROM 'ZZZKIOMZZZ') FROM DUAL;
SELECT TRIM(LEADING 'Z' FROM 'ZZZZKOINKLDZZZ') FROM DUAL;
SELECT TRIM(TRAILING 'Z' FROM 'ZZKDJFALZZZZZZ') FROM DUAL;

/*

    LOWER / UPPER / INITCAP

*/

SELECT LOWER ('WELCOME HOME') FROM DUAL;
SELECT UPPER('helloworld') FROM DUAL;
SELECT INITCAP('hello world') FROM DUAL;


/*

    CONCAT
    
*/

SELECT '가나다'||'라마바' FROM DUAL;

SELECT CONCAT('가나다', '라마바')FROM DUAL;

/*

    REPLACE
    
*/

SELECT REPLACE('경기도 광명시 광명동', '광명동', '소하동') FROM DUAL;

/*

    숫자 관련

SELECT MOD(10, 3) FROM DUAL;
SELECT MOD(-10, 3) FROM DUAL;

SELECT CEIL(123.1235463135215) FROM DUAL;

SELECT TRUNC(123.456874642, 1) FROM DUAL;


/*

    날짜 관련

*/


SELECT ROUND(SYSDATE - HIRE_DATE) || '일' 근무일수,
        FLOOR(MONTHS_BETWEEN(SYSDATE, HIRE_DATE)) || '개월' 근무개월수
FROM EMPLOYEE;

SELECT ADD_MONTHS(SYSDATE, 5) FROM DUAL;


SELECT EMP_NAME, HIRE_DATE, ADD_MONTHS(HIRE_DATE, 6)
FROM EMPLOYEE;


SELECT NEXT_DAY(SYSDATE, '토요일') FROM DUAL;
